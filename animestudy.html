import pandas as pd
import numpy as np
from matplotlib import pyplot as plt
from matplotlib.font_manager import FontProperties

plot_count = 1
save_path = "2016summer.csv"
df = pd.read_csv(save_path)
raw_table = pd.crosstab(df["ep"],df["anime_name"],df['download_num'].values,aggfunc='sum')
raw_table = raw_table[1:]
per_table = raw_table/raw_table.ix[1,:]
increase_table = (raw_table.ix[2:12,:] - raw_table.ix[1:11,:].values)/raw_table.ix[1:11,:].values

for table in [raw_table,per_table,increase_table]:
    plt.figure('%d' % plot_count,figsize=(15,8))
    nrow,ncol = table.shape
    X = table.index.values
    colnames = table.columns
    for col in range(ncol):
        Y = table.values[:,col]
        plt.plot(X,Y,label = colnames[col])
    plt.legend(loc='center left', bbox_to_anchor=(1, 0.5))
    plt.xticks(X,X)
    if table is per_table:
        plt.yticks(np.arange(0,15)/10.)
    plt.title('animestudy')
    plt.grid()
    plt.show()
    plot_count += 1
